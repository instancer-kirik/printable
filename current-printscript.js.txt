    // ---------- Print settings (label size / DPI) ----------

    defaultPrintSettings() {
      return {
        dpi: 203, // 203 or 300 - unknown until confirmed on the ZT231; use "Print Config Label" to check
        widthIn: 1.5, // physical label width in inches (ZT231 stock: 1.5" x 1")
        heightIn: 1.0, // physical label height in inches
        nameFontDots: 22,
        priceFontDots: 45,
        storeName: "VDG",      // store name printed at top of label
        // --- Physical printer adjustments ---
        labelShift: 0,         // ^LS  horizontal offset in dots (+ = right, − = left). Tune this to align label — each dot = ~0.005" at 203dpi
        labelTop: 0,           // ^LT  vertical offset in dots   (+ = down,  - = up)
        darkness: 18,          // ~SD  print darkness 0-30
        printSpeed: 4,         // ^PR  print speed in inches/sec (1-14 depending on model)
        // --- Barcode ---
        storeNameFontDots: 26,  // font size for store name (VDG) header
        barcodeModuleWidth: 1,   // ^BY module width — integers only (ZPL rounds decimals); 1 fits better on 1.5" label
        barcodeHeightDots: 40,  // barcode bar height in dots (not counting HRI text below)
        barcodeXOffset: 0,      // manual horizontal nudge in dots (+ = right, − = left)
        // --- AI Pricing ---
        pricingFactor: 0.85,    // fraction of market high price to suggest (e.g. 0.85 = 15% below retail)
      };
    }

    loadPrintSettings() {
      try {
        const saved = localStorage.getItem("pm-print-settings-v3");
        if (saved) {
          return Object.assign(this.defaultPrintSettings(), JSON.parse(saved));
        }
      } catch (error) {
        log("Could not load print settings, using defaults:", error);
      }
      return this.defaultPrintSettings();
    }

    loadMissedItems() {
      try {
        const saved = localStorage.getItem("pm-missed-items");
        return saved ? JSON.parse(saved) : [];
      } catch { return []; }
    }

    saveMissedItems() {
      try {
        localStorage.setItem("pm-missed-items", JSON.stringify(this.missedItems));
      } catch (e) { log("Could not save missed items:", e); }
    }

    addMissedItem(upc) {
      if (this.missedItems.some(m => m.upc === upc)) return; // no duplicates
      this.missedItems.push({ upc, timestamp: Date.now() });
      this.saveMissedItems();
      this.updateMissedBadge();
    }

    makeDraggable(element) {
      let pos1 = 0,
        pos2 = 0,
        pos3 = 0,
        pos4 = 0;
      const header = element.querySelector(".header");

      header.onmousedown = dragMouseDown;

      function dragMouseDown(e) {
        e.preventDefault();
        pos3 = e.clientX;
        pos4 = e.clientY;
        document.onmouseup = closeDragElement;
        document.onmousemove = elementDrag;
      }

      function elementDrag(e) {
        e.preventDefault();
        pos1 = pos3 - e.clientX;
        pos2 = pos4 - e.clientY;
        pos3 = e.clientX;
        pos4 = e.clientY;
        element.style.top = element.offsetTop - pos2 + "px";
        element.style.left = element.offsetLeft - pos1 + "px";
      }

      function closeDragElement() {
        document.onmouseup = null;
        document.onmousemove = null;
      }
    }

    findUpcField() {
      // Case 1: placeholder explicitly says "Enter UPC"
      let field = document.querySelector('input[placeholder="Enter UPC" i]');
      if (field) return field;

      // Case 2: generic placeholder, but a <mat-label> nearby contains "upc"
      const labels = document.querySelectorAll('mat-label');
      for (const label of labels) {
        if (label.textContent.trim().toLowerCase() === 'upc') {
          const formField = label.closest('mat-form-field');
          if (formField) {
            const input = formField.querySelector('input');
            if (input) return input;
          }
        }
      }

      return null;
    }

    getCurrentProductData() {
      const data = {};

      try {
        // Get UPC
        const upcField = this.findUpcField();
        data.upc = upcField?.value || "";

        // Get description/name
        // The field is an <input class="app-text-input"> with placeholder "Enter product description"
        // mat-input IDs like "mat-input-1" are dynamic, so match by placeholder or class+label
        const nameField =
          document.querySelector('input[placeholder="Enter product description"]') ||
          document.querySelector('textarea[name="stringAreaCtrl"]') ||
          document.querySelector("string-area-picker textarea") ||
          this.findFieldByLabelText("Description") ||
          this.findFieldByLabelText("Name");
        data.name = nameField?.value || "";

        // Get price
        const priceField =
          document.querySelector("#unit-price") ||
          document.querySelector('money-picker input[id="unit-price"]') ||
          this.findFieldByLabelText("Price");
        data.price = priceField?.value || "";

        // Get cost
        const costField =
          document.querySelector("#cost") ||
          document.querySelector('money-picker input[id="cost"]') ||
          this.findFieldByLabelText("Cost");
        data.cost = costField?.value || "";

        // Get department
        const deptElement = document.querySelector("departments-picker");
        data.department = deptElement?.textContent?.trim() || "";

        log("Current product data:", data);
        return data;
      } catch (error) {
        log("Error getting current product data:", error);
        return data;
      }
    }

    // Checks whether the page is showing the "Product not found with this UPC" alert
    isProductNotFound() {
      const spans = document.querySelectorAll('app-alerts .alert-item span');
      for (const span of spans) {
        if (span.textContent.includes('Product not found with this UPC')) return true;
      }
      return false;
    }

    // Normalise a UPC string for loose comparison:
    // strip leading zeros and non-digits so "012345678901" ≡ "12345678901".
    _normaliseUpc(upc) {
      return (upc || "").replace(/\D/g, "").replace(/^0+/, "") || "";
    }

    async pollForProductData(expectedUpc, maxAttempts = 15) {
      log(`Starting to poll for product data for UPC: ${expectedUpc}`);
      const normExpected = this._normaliseUpc(expectedUpc);

      for (let attempt = 1; attempt <= maxAttempts; attempt++) {
        // Bail early if the system already told us the product doesn't exist
        if (this.isProductNotFound()) {
          log(`🔍 "Product not found" alert detected on attempt ${attempt} — stopping poll`);
          return null;
        }

        const currentData = this.getCurrentProductData();
        const formUpc     = (currentData.upc || "").trim();
        const normForm    = this._normaliseUpc(formUpc);

        log(
          `Poll attempt ${attempt}/${maxAttempts}: UPC="${formUpc}", Name="${currentData.name}", Price="${currentData.price}"`,
        );

        // UPC match check — lenient: ignore leading-zero differences,
        // and treat empty form UPC as "still loading" (not a mismatch).
        const upcOk = !normForm || normForm === normExpected;

        if (!upcOk) {
          log(`⏳ Stale UPC in form (${formUpc} ≠ ${expectedUpc}), waiting for form to update…`);
          this.updateStatus(`Loading... (${attempt}/${maxAttempts})`);
          await sleep(250);
          continue;
        }

        // Primary signal: name field populated means the app has loaded the product.
        // Price may arrive at the same time or slightly after; we return as soon
        // as we see a name so the user isn’t kept waiting unnecessarily.
        // printCurrentProduct will block + warn if price is still missing.
        if (currentData.name) {
          log(`✅ Product name detected on attempt ${attempt} — returning data (price="${currentData.price}")`);
          return currentData;
        }

        this.updateStatus(`Loading... (${attempt}/${maxAttempts})`);
        await sleep(250);
      }

      log(`❌ No product data found after ${maxAttempts} attempts`);
      return null;
    }

    // Strip currency symbols/spaces and check the number is > 0.
    isValidPrice(price) {
      const n = parseFloat((price || "").toString().replace(/[^0-9.]/g, ""));
      return !isNaN(n) && n > 0;
    }

    // fetch() is subject to the page's CSP connect-src, which blocks localhost
    // on marktpos. GM_xmlhttpRequest runs outside the page context and isn't
    // subject to that CSP, so we use it for all localhost printer/server calls.
    gmRequest(details) {
      return new Promise((resolve, reject) => {
        if (typeof GM_xmlhttpRequest !== "function") {
          reject(new Error("GM_xmlhttpRequest not available - re-check @grant in script header"));
          return;
        }
        GM_xmlhttpRequest({
          ...details,
          onload: (response) => resolve(response),
          onerror: (response) => reject(new Error(
            `Network error reaching ${details.url} (status ${response.status || 0}) — ` +
            (response.statusText || "connection refused or cert rejected")
          )),
          ontimeout: () => reject(new Error("Request timed out")),
        });
      });
    }

    // Always use the plain-HTTP port regardless of page protocol.
    // GM_xmlhttpRequest runs in a privileged context and is not subject to
    // mixed-content restrictions, so this works fine on HTTPS pages.
    // Port 9101 (HTTPS) is avoided because Tampermonkey does not bypass
    // the agent's self-signed cert, causing status-0 errors.
    // Chrome's Private Network Access (PNA) policy also blocks window.BrowserPrint
    // SDK calls to localhost from HTTPS pages, so we skip the SDK entirely.
    get bpBaseUrl() {
      return "http://localhost:9100";
    }


    findFieldByLabelText(labelText) {
      const labels = document.querySelectorAll("mat-label");
      const target = labelText.trim().toLowerCase();
      for (const label of labels) {
        if (label.textContent.trim().toLowerCase().startsWith(target)) {
          const formField = label.closest("mat-form-field");
          if (formField) {
            const field = formField.querySelector("input, textarea");
            if (field) return field;
          }
        }
      }
      return null;
    }

    async printCurrentProduct(productData = null, qty = null) {
      const count = (qty != null) ? qty : this.printQty;
      this.updateStatus(count > 1 ? `Printing ${count} labels...` : "Printing...");

      // Use provided data or get from UI as fallback
      const dataToUse = productData || this.getCurrentProductData();

      if (!this.isValidPrice(dataToUse.price)) {
        this.updateStatus("⚠️ Cannot print: No price set for this product");
        log("Print blocked: Product has no valid price");
        return;
      }

      try {
        // Use Zebra APIs directly
        log(`🦓 Printing via Zebra APIs (qty=${count})`);
        await this.printViaZebra(dataToUse, count);
        this.updateStatus(count > 1 ? `✅ ${count} labels printed!` : "✅ Label printed successfully!");
      } catch (zebraError) {
        log("Zebra printing failed:", zebraError);

    // Send raw ZPL via the Browser Print local agent REST API.
    async sendZPL(zplCommands) {
      const device = await this.getSelectedPrinterDevice();
      log("✅ Sending ZPL to printer:", device.name || device.uid);
      const response = await this.gmRequest({
        method: "POST",
        url: `${this.bpBaseUrl}/write`,
        data: JSON.stringify({ device, data: zplCommands }),
        headers: { "Content-Type": "application/json" },
        timeout: 10000,
      });
      if (response.status >= 200 && response.status < 300) {
        log("✅ ZPL accepted by Browser Print agent");
      } else {
        throw new Error(`Browser Print agent returned ${response.status}: ${response.responseText}`);
      }
    }

    // List printers via the Browser Print local agent REST API (port 9100, plain HTTP).
    // GM_xmlhttpRequest bypasses mixed-content and PNA restrictions, so this works
    // from any page including HTTPS. We skip window.BrowserPrint because Chrome's
    // Private Network Access policy blocks its XHR calls to localhost on HTTPS pages.
    async listBrowserPrintPrinters() {
      const response = await this.gmRequest({
        method: "GET",
        url: `${this.bpBaseUrl}/available?type=printer`,
        timeout: 5000,
      });
      if (response.status < 200 || response.status >= 300) {
        throw new Error(`Browser Print agent unavailable (HTTP ${response.status})`);
      }
      const body = JSON.parse(response.responseText || "{}");
      return Array.isArray(body) ? body : body.printer || body.Printer || [];
    }

    // Resolve which printer device to send to: saved preference > ZT231 by name > sole device.
    // Result is cached so every print doesn't pay a full HTTP round-trip to the agent.
    async getSelectedPrinterDevice() {
      if (this.cachedPrinterDevice) return this.cachedPrinterDevice;
      const printers = await this.listBrowserPrintPrinters();
      if (!printers || printers.length === 0) {
        throw new Error("No printers found by Browser Print");
      }

      const savedUid = localStorage.getItem("pm-selected-printer-uid");
      if (savedUid) {
        const match = printers.find((p) => p.uid === savedUid);
        if (match) { this.cachedPrinterDevice = match; return match; }
        log(`⚠️ Saved printer uid "${savedUid}" not found among current devices`);
      }

      const zt231 = printers.find((p) =>
        (p.name || "").toUpperCase().includes("ZT231"),
      );
      if (zt231) {
        log("🎯 Auto-matched ZT231 printer:", zt231);
        this.cachedPrinterDevice = zt231;
        return zt231;
      }

      if (printers.length === 1) {
        this.cachedPrinterDevice = printers[0];
        return printers[0];
      }

      throw new Error(
        `Multiple printers found (${printers.map((p) => p.name).join(", ")}) - use "Select Printer" to choose one`,
      );
    }

    // Returns the store logo as a ZPL ^GFA graphic command, or "" if none configured.
    // To add your logo: convert it to ZPL using Zebra's ZebraDesigner / online
    // "image to ZPL" tools set to your printer's DPI, then paste the resulting
    // ^GFA,... line as the value of window.PM_STORE_LOGO_ZPL before this script
    // runs, or hardcode it here. Placement (x,y) is controlled by logoX/logoY below.
    getLogoZPL() {
      const logoData = window.PM_STORE_LOGO_ZPL || this.printSettings.logoZPL || "";
      if (!logoData) return "";
      const logoX = this.printSettings.logoX ?? 20;
      const logoY = this.printSettings.logoY ?? 5;
      return `^FO${logoX},${logoY}${logoData}^FS\n`;
    }

    // Wraps a product name into at most 2 lines, each at most maxChars characters.
    // Breaks on word boundaries; hard-truncates only if a single word exceeds maxChars.
    wrapNameLines(name, maxChars) {
      if (!name) return { line1: "", line2: "" };
      if (name.length <= maxChars) return { line1: name, line2: "" };

      const words = name.split(/\s+/);
      let line1 = "";
      let splitAt = 0;

      for (let i = 0; i < words.length; i++) {
        const candidate = line1 ? `${line1} ${words[i]}` : words[i];
        if (candidate.length <= maxChars) {
          line1 = candidate;
          splitAt = i + 1;
        } else {
          break;
        }
      }

      // Edge case: first word alone exceeds maxChars — hard-break it
      if (!line1) {
        line1 = words[0].substring(0, maxChars);
        splitAt = 1;
      }

      // Build line2 from remaining words, fitting whole words only
      const restWords = words.slice(splitAt);
      let line2 = "";
      for (const w of restWords) {
        const candidate = line2 ? `${line2} ${w}` : w;
        if (candidate.length <= maxChars) line2 = candidate;
        else break;
      }
      // If still nothing fits (single long word), hard-truncate
      if (!line2 && restWords.length) line2 = restWords[0].substring(0, maxChars);

      return { line1, line2 };
    }

    createZPLCommands(productData, qty = 1) {
      const name = (productData.name || "NO NAME").trim();
      const price = productData.price || "0.00";
      const upc = productData.upc || "";
      const cleanPrice = price.toString().replace(/[^0-9.]/g, "");
      const formattedPrice = parseFloat(cleanPrice || "0").toFixed(2);

      const { widthDots, heightDots } = this.getLabelDots();
      const nameFont = this.printSettings.nameFontDots;
      const priceFont = this.printSettings.priceFontDots;
      const storeName = (this.printSettings.storeName || "").trim();
      const storeFont = this.printSettings.storeNameFontDots ?? 26;

      // ZPL commands sized from configured label dimensions + DPI (see printSettings)
      // ~SD must precede ^XA so the printer applies it before rendering the format
      let zpl = `~SD${String(this.printSettings.darkness ?? 15).padStart(2, "0")}\n`; // Darkness (immediate)
      zpl += "^XA\n"; // Start format
      zpl += `^PR${this.printSettings.printSpeed ?? 4}\n`;  // Print speed
      // Always send ^LS and ^LT so our setting is authoritative every job.
      // labelShift compensates for physical media-to-printhead offset (+ shifts print right).
      const ls = this.printSettings.labelShift ?? 0;
      // ^LH must come before ^LS — on many firmware versions ^LH resets the label shift, cancelling ^LS.
      zpl += "^LH0,0\n"; // Set label home position
      zpl += `^LS${ls}\n`;
      if (this.printSettings.labelTop) zpl += `^LT${this.printSettings.labelTop}\n`;
      // ^PW must cover widthDots + any positive ^LS shift, otherwise ^LS clips the right edge of centered content.
      zpl += "^PW" + (widthDots + Math.max(0, ls)) + "\n"; // Print width in dots
      zpl += "^LL" + heightDots + "\n"; // Set label length in dots

      // Optional store logo
      zpl += this.getLogoZPL();

      const fieldW = widthDots - 20; // 10-dot inset each side

      // Barcode sizing: auto-calculate module width to fill ~85% of fieldW.
      // Code 128C for a 12-digit UPC = ~121 modules (bars + quiet zones).
      // ZPL ^BY only accepts integers, so we floor to the nearest whole dot.
      const HRI_FONT         = 14;   // small fixed font for HRI — separate from storeFont
      const barcodeHeight    = this.printSettings.barcodeHeightDots  ?? 40;
      let barcodeModuleWidth = Math.round(this.printSettings.barcodeModuleWidth ?? 2); // ^BY only accepts integers
      // UPC-A (12 digits): 95 symbol + 18 quiet zone = 113 modules
      // Code 128C fallback: 101 symbol + 20 quiet zone = 121 modules
      // A 13-digit UPC starting with 0 is an EAN-13 encoding of a UPC-A — strip the leading 0.
      const upcNorm = /^0\d{12}$/.test(upc) ? upc.substring(1) : upc;
      const isUpcA = /^\d{12}$/.test(upcNorm);
      log(`🔖 Barcode type: ${isUpcA ? 'UPC-A (^BU)' : 'Code 128 (^BC)'} for UPC="${upc}" → norm="${upcNorm}"`);
      const TOTAL_MODULES   = isUpcA ? 113 : 121;
      // Auto-clamp module width so the barcode never overdraws the label
      while (barcodeModuleWidth > 1 && barcodeModuleWidth * TOTAL_MODULES > fieldW) barcodeModuleWidth--;
      const estBarcodeWidth = barcodeModuleWidth * TOTAL_MODULES;

      // --- Y layout: name block at top, price+barcode anchored from bottom ---
      const topMargin  = 15;
      const storeNameY = topMargin;
      const nameY1     = storeNameY + storeFont + 4;
      const nameLineH  = Math.round(nameFont * 1.2);
      const nameY2     = nameY1 + nameLineH;
      // Barcode sits at bottom; HRI_FONT (not storeFont) is the HRI text height
      const barcodeY      = heightDots - barcodeHeight - HRI_FONT - 3;
      const priceYBottom  = barcodeY - priceFont - 6;          // ideal: just above barcode
      const priceYTop     = nameY2 + nameFont + 8;             // minimum: 8 dots below name block
      const priceY        = Math.max(priceYBottom, priceYTop); // whichever needs more room

      const maxCharsPerLine = Math.max(8, Math.floor(fieldW / (nameFont * 0.65)));
      const { line1, line2 } = this.wrapNameLines(name, maxCharsPerLine);

      // Store name — centered
      if (storeName) {
        zpl += `^CF0,${storeFont}\n`;
        zpl += `^FO10,${storeNameY}^FB${fieldW},1,0,C,0^FD${storeName}^FS\n`;
      }

      // Product name — centered, 1 or 2 lines
      zpl += `^CF0,${nameFont}\n`;
      zpl += `^FO10,${nameY1}^FB${fieldW},1,0,C,0^FD${line1}^FS\n`;
      if (line2) {
        zpl += `^FO10,${nameY2}^FB${fieldW},1,0,C,0^FD${line2}^FS\n`;
      }

      // Price — centered
      zpl += `^CF0,${priceFont}\n`;
      zpl += `^FO10,${priceY}^FB${fieldW},1,0,C,0^FD$${formattedPrice}^FS\n`;

      if (upc && upc.length >= 8) {
        const barcodeX = Math.max(0, Math.round((widthDots - estBarcodeWidth) / 2) + (this.printSettings.barcodeXOffset ?? 0));
        zpl += `^CF0,${HRI_FONT}\n`;
        // ^BY must precede ^FO — placed after ^FO the printer ignores it and uses its stored default module width.
        zpl += `^BY${barcodeModuleWidth}\n`;
        if (isUpcA) {
          // ^BU expects 11 digits — printer calculates and appends the 12th (check digit)
          // Sending all 12 digits with e=N produces a garbled barcode; this is the correct form
          zpl += `^FO${barcodeX},${barcodeY}^BUN,${barcodeHeight},Y,N,Y^FD${upcNorm.substring(0, 11)}^FS\n`;
        } else {
          // Code 128 for non-UPC-A (EAN-13, UPC-E, etc.)
          zpl += `^FO${barcodeX},${barcodeY}^BCN,${barcodeHeight},Y,N,N^FD${upcNorm}^FS\n`;
        }
      }

      // Print quantity — ^PQ tells the printer to produce N copies from one job
      if (qty > 1) zpl += `^PQ${qty}\n`;

      zpl += "^XZ\n"; // End format

      log(`🏷️ Generated ZPL (${widthDots}x${heightDots} dots @ ${this.printSettings.dpi}dpi, qty=${qty}):`, zpl);
      return zpl;
    }

    createCustomZPL(productData, settings) {
      const name = (productData.name || "NO NAME").trim();
      const price = productData.price || "0.00";
      const upc = productData.upc || "";
      const cleanPrice = price.toString().replace(/[^0-9.]/g, "");
      const formattedPrice = parseFloat(cleanPrice || "0").toFixed(2);

      const dpi = parseInt(settings.dpi, 10) || this.printSettings.dpi;
      const widthDots = Math.round(parseFloat(settings.width) * dpi);
      const heightDots = Math.round(parseFloat(settings.height) * dpi);

      // Custom ZPL with user settings - always 2 name lines
      // ~SD must precede ^XA
      let zpl = `~SD${String(parseInt(settings.darkness ?? this.printSettings.darkness ?? 15, 10)).padStart(2, "0")}\n`;
      zpl += "^XA\n";
      zpl += `^PR${settings.printSpeed ?? this.printSettings.printSpeed ?? 4}\n`;
      const ls = settings.labelShift ?? this.printSettings.labelShift ?? 0;
      const lt = settings.labelTop   ?? this.printSettings.labelTop   ?? 0;
      // ^LH must come before ^LS — on many firmware versions ^LH resets the label shift, cancelling ^LS.
      zpl += "^LH0,0\n";
      zpl += `^LS${ls}\n`;
      if (lt) zpl += `^LT${lt}\n`;
      // ^PW must cover widthDots + any positive ^LS shift, otherwise ^LS clips the right edge of centered content.
      zpl += "^PW" + (widthDots + Math.max(0, ls)) + "\n";
      zpl += "^LL" + heightDots + "\n";

      zpl += this.getLogoZPL();

      const storeName  = (this.printSettings.storeName || "").trim();
      const storeFont  = parseInt(settings.storeNameFontDots ?? this.printSettings.storeNameFontDots ?? 26, 10);
      const fieldWc    = widthDots - 20;
      const priceSize  = parseInt(settings.priceSize, 10);
      const nameSize   = parseInt(settings.nameSize, 10);

      const HRI_FONT         = 14;
      const barcodeHeight    = parseInt(settings.barcodeHeightDots ?? this.printSettings.barcodeHeightDots ?? 40, 10);
      let barcodeModuleWidth = Math.round(parseFloat(settings.barcodeModuleWidth ?? this.printSettings.barcodeModuleWidth ?? 2));
      // A 13-digit UPC starting with 0 is an EAN-13 encoding of a UPC-A — strip the leading 0.
      const upcNorm = /^0\d{12}$/.test(upc) ? upc.substring(1) : upc;
      const isUpcA = /^\d{12}$/.test(upcNorm);
      log(`🔖 Barcode type: ${isUpcA ? 'UPC-A (^BU)' : 'Code 128 (^BC)'} for UPC="${upc}" → norm="${upcNorm}"`);
      const TOTAL_MODULES   = isUpcA ? 113 : 121;
      // Auto-clamp module width so the barcode never overdraws the label
      while (barcodeModuleWidth > 1 && barcodeModuleWidth * TOTAL_MODULES > fieldWc) barcodeModuleWidth--;
      const estBarcodeWidth = barcodeModuleWidth * TOTAL_MODULES;

      const topMargin  = 15;
      const storeNameY = topMargin;
      const nameY1c    = storeNameY + storeFont + 4;
      const nameLineHc = Math.round(nameSize * 1.2);
      const nameY2c    = nameY1c + nameLineHc;
      const barcodeY      = heightDots - barcodeHeight - HRI_FONT - 3;
      const priceYBottom  = barcodeY - priceSize - 6;
      const priceYTop     = nameY2c + nameSize + 8;
      const priceYc       = Math.max(priceYBottom, priceYTop);

      const maxCharsC = Math.max(8, Math.floor(fieldWc / (nameSize * 0.65)));
      const { line1, line2 } = this.wrapNameLines(name, maxCharsC);

      if (storeName) {
        zpl += `^CF0,${storeFont}\n`;
        zpl += `^FO10,${storeNameY}^FB${fieldWc},1,0,C,0^FD${storeName}^FS\n`;
      }

      zpl += `^CF0,${nameSize}\n`;
      zpl += `^FO10,${nameY1c}^FB${fieldWc},1,0,C,0^FD${line1}^FS\n`;
      if (line2) {
        zpl += `^FO10,${nameY2c}^FB${fieldWc},1,0,C,0^FD${line2}^FS\n`;
      }
      zpl += `^CF0,${priceSize}\n`;
      zpl += `^FO10,${priceYc}^FB${fieldWc},1,0,C,0^FD$${formattedPrice}^FS\n`;

    setupBarcodeDetection() {
      const trySetup = () => {
        const upcField = this.findUpcField();

        // Rebind whenever the field reference changes (Angular route navigation gives a new element)
        if (upcField && upcField !== this.boundUpcField) {
          this.boundUpcField = upcField;
          log("✅ Binding barcode detection to UPC field:", upcField);

        // Monitor for rapid input (typical of barcode scanners)
        upcField.addEventListener("input", (e) => {
          const currentValue = e.target.value;
          log(`📝 UPC field input detected: "${currentValue}"`);

          // Skip if already processing
          if (this.processingBarcode) {
            log(`🚫 Already processing barcode, skipping input event`);
            return;
          }

          // Clear previous timeout
          if (this.scanTimeout) {
            clearTimeout(this.scanTimeout);
          }

          // Set timeout to detect end of scan
          this.scanTimeout = setTimeout(() => {
            log(
              `⏱️ Input timeout reached, processing barcode: "${currentValue}"`,
            );
            this.handleBarcodeInput(currentValue);
          }, 300); // Increased to 300ms for Windows compatibility
        });

        // Also monitor for Enter key (some scanners send Enter)
        upcField.addEventListener("keydown", (e) => {
          if (e.key === "Enter") {
            log(`⌨️ Enter key detected in UPC field`);
            e.preventDefault();
            const currentValue = e.target.value;

            // Skip if already processing
            if (this.processingBarcode) {
              log(`🚫 Already processing barcode, skipping Enter event`);
              return;
            }

            // Clear timeout since we're handling immediately
            if (this.scanTimeout) {
              clearTimeout(this.scanTimeout);
            }
            setTimeout(() => {
              log(`⚡ Processing barcode via Enter key: "${currentValue}"`);
              this.handleBarcodeInput(currentValue);
            }, 100);
          }
        });

          // Remove change event - was causing double prints on refocus

          return true;
        }
        return false;
      };

      trySetup();

      // Watch for the UPC field to appear (initial load) or change (Angular route navigation).
      // MutationObserver catches DOM changes; heartbeat catches cases where the element
      // reference silently changes without a DOM mutation (Angular component reuse).
      const observer = new MutationObserver(() => { trySetup(); });
      observer.observe(document.body, { childList: true, subtree: true });
      setInterval(() => { trySetup(); }, 2000);
    }

    async handleBarcodeInput(upcValue) {
      const now = Date.now();
      const recentDup = upcValue === this.lastUpcValue && (now - this.lastUpcTime) < 1500;
      if (!upcValue || recentDup) {
        log(`Skipping barcode input: empty=${!upcValue}, recentDup=${recentDup}`);
        return;
      }
      this.lastUpcTime = now;

      // Prevent concurrent processing
      if (this.processingBarcode) {
        log(`🚫 Already processing barcode: ${upcValue}, skipping`);
        return;
      }

      // Set processing flag and last UPC immediately
      this.processingBarcode = true;
      this.lastUpcValue = upcValue;
      log(
        `🔍 Barcode detected: ${upcValue}, Auto-print enabled: ${this.autoPrintEnabled}`,
      );

      try {
        // Check if this looks like a barcode (all digits, reasonable length)
        if (/^\d{8,14}$/.test(upcValue)) {
          this.updateStatus(`Scanned: ${upcValue}`);

          // Fast path: same product rescanned — reuse cached data so we don't
          // clear the fields and then poll for data the app won't re-fetch.
          const cachedMatch = this.lastProductData &&
            this._normaliseUpc(this.lastProductData.upc) === this._normaliseUpc(upcValue);

          if (cachedMatch) {
            log(`🔄 Repeat scan of same product — using cached data`);
            this.updateStatus(`🔄 ${this.lastProductData.name}`, "success");
            if (this.autoPrintEnabled) {
              log(`🖨️ Auto-print (repeat) printing (qty=${this.printQty})…`);
              try {
                await this.printCurrentProduct(this.lastProductData, this.printQty);
                log("✅ Auto-print (repeat) completed");
              } catch (error) {
                log(`❌ Auto-print (repeat) failed:`, error);
                this.updateStatus(`Print failed: ${error.message}`);
              }
            }
            return;
          }

          // Clear existing product data from UI fields
          this.clearProductFields();

          // Give the system time to process the scan before polling
          await sleep(300);

          // Poll for product data to load
          const currentData = await this.pollForProductData(upcValue);

          if (currentData) {
            // Check if this is a different product than last scan BEFORE clearing
            const isNewProduct =
              !this.lastProductData ||
              this.lastProductData.upc !== currentData.upc ||
              this.lastProductData.name !== currentData.name;

            // Now update the stored product data
            this.lastProductData = currentData;

            if (isNewProduct) {
              this.updateStatus(`✅ NEW: ${currentData.name}`, "success");
              log(
                `✅ NEW product loaded: Name="${currentData.name}", Price="${currentData.price}", UPC="${currentData.upc}"`,
              );
            } else {
              this.updateStatus(`🔄 SAME: ${currentData.name}`, "success");
              log(
                `🔄 Same product rescanned: Name="${currentData.name}", Price="${currentData.price}", UPC="${currentData.upc}"`,
              );
            }

            // Auto-print if enabled — fires on every accepted scan (dedup handled by 1.5s time gate above)
            if (this.autoPrintEnabled) {
              log(`🖨️ Auto-print printing (qty=${this.printQty})…`);
              try {
                await this.printCurrentProduct(currentData, this.printQty);
                log("✅ Auto-print completed successfully");
              } catch (error) {
                log(`❌ Auto-print failed:`, error);
                this.updateStatus(`Print failed: ${error.message}`);
              }
            }
          } else {
            this.lastProductData = null;
            if (this.isProductNotFound()) {
              this.addMissedItem(upcValue);
              log(`🔍 Out-of-system item: ${upcValue} — added to missed list`);
              this.updateStatus(`⚠️ Not in system — added to missed list (${this.missedItems.length})`);
            } else {
              this.updateStatus(`❌ No data for UPC: ${upcValue}`);
              log(`❌ No data found for UPC: ${upcValue}, auto-print will be skipped`);
            }
          }
        } else {
          this.updateStatus(`❌ Invalid barcode format: ${upcValue}`);
          this.processingBarcode = false;
        }
      } catch (error) {
        log("Error in handleBarcodeInput:", error);
        this.updateStatus("Error processing barcode");
      } finally {
        // Always clear processing flag when done
        this.processingBarcode = false;
      }
    }

    // Tries Open Food Facts first (free, no key), falls back to UPCitemdb trial.
    async lookupProductByUPC(upc) {
      // Open Food Facts
      try {
        const res = await this.gmRequest({
          method: 'GET',
          url: `https://world.openfoodfacts.org/api/v2/product/${upc}.json?fields=product_name,brands,categories,quantity`,
          headers: { 'User-Agent': 'MarktPOS-Lookup/1.0' },
          timeout: 8000,
        });
        if (res.status === 200) {
          const d = JSON.parse(res.responseText);
          if (d.status === 1 && d.product) {
            const p = d.product;
            return {
              source: 'Open Food Facts',
              name: p.product_name || '',
              brand: p.brands || '',
              category: (p.categories || '').split(',')[0].trim(),
              quantity: p.quantity || '',
            };
          }
        }
      } catch (e) { log('Open Food Facts lookup failed:', e); }

      // UPCitemdb trial fallback
      try {
        const res = await this.gmRequest({
          method: 'GET',
          url: `https://api.upcitemdb.com/prod/trial/lookup?upc=${upc}`,
          timeout: 8000,
        });
        if (res.status === 200) {
          const d = JSON.parse(res.responseText);
          if (d.code === 'OK' && d.items?.length) {
            const item = d.items[0];
            return {
              source: 'UPCitemdb',
              name: item.title || '',
              brand: item.brand || '',
              category: item.category || '',
              quantity: item.size || '',
              priceMin: item.lowest_recorded_price  || null,
              priceMax: item.highest_recorded_price || null,
            };
          }
        }
      } catch (e) { log('UPCitemdb lookup failed:', e); }

      return null;
    }

          ${sec('📦 Product Info',
            `<div id="pm-r-lookup-status" style="font-size:12px;color:#aaa;margin-bottom:8px;">⏳ Looking up product…</div>
             ${row('Name', inp('pm-r-name','text','Product name'))}
             <div style="display:flex;gap:8px;">
               <label style="flex:1;font-size:12px;color:#444;">Brand
                 ${inp('pm-r-brand','text','Brand')}</label>
               <label style="flex:1;font-size:12px;color:#444;">Size / Qty
                 ${inp('pm-r-size','text','e.g. 12 oz')}</label>
             </div>`
          )}

          ${sec('💰 AI Pricing',
            `<div id="pm-r-market-row" style="display:none;background:#f0f7ff;border-radius:5px;padding:8px 10px;margin-bottom:10px;">
               <span style="font-size:12px;color:#444;">Market range: </span>
               <strong id="pm-r-market-range" style="font-size:13px;"></strong>
               <span style="font-size:12px;color:#666;"> &nbsp;·&nbsp; Suggested at
                 <input id="pm-r-factor" type="number" min="0.1" max="3" step="0.05" value="${factor}"
                   style="width:52px;padding:2px 4px;font-size:12px;border:1px solid #bbb;border-radius:3px;text-align:center;"> ×
               </span>
               <button id="pm-r-apply-ai"
                 style="display:inline-block;margin-left:6px;padding:3px 10px;font-size:12px;background:#198754;color:white;border:none;border-radius:4px;cursor:pointer;">
                 Use <span id="pm-r-ai-price"></span> ✓
               </button>
             </div>
             <div style="display:flex;gap:8px;align-items:flex-end;">
               <label style="flex:1;font-size:12px;color:#444;">Your Price ($)
                 <input id="pm-r-price" type="number" step="0.01" min="0" placeholder="0.00"
                   style="width:100%;box-sizing:border-box;padding:7px 8px;border:2px solid #28a745;border-radius:4px;font-size:16px;font-weight:700;margin-top:3px;"></label>
               <label style="flex:1;font-size:12px;color:#444;">Cost ($) <em style="font-size:10px;color:#aaa;">(optional)</em>
                 <input id="pm-r-cost" type="number" step="0.01" min="0" placeholder="0.00"
                   style="width:100%;box-sizing:border-box;padding:7px 8px;border:1px solid #ccc;border-radius:4px;font-size:14px;margin-top:3px;"></label>
             </div>
             <div id="pm-r-margin" style="font-size:12px;color:#555;margin-top:7px;min-height:18px;"></div>`
          )}

          <details style="margin-bottom:10px;">
            <summary style="font-size:12px;font-weight:700;color:#555;text-transform:uppercase;letter-spacing:.6px;cursor:pointer;user-select:none;">
              🔍 Market Research Links
            </summary>
            <div style="display:flex;flex-wrap:wrap;gap:6px;margin-top:8px;">
              <a href="https://www.google.com/search?q=${q}" target="_blank" style="${linkStyle}">Google</a>
              <a href="https://www.amazon.com/s?k=${q}" target="_blank" style="${linkStyle}">Amazon</a>
              <a href="https://www.walmart.com/search?q=${q}" target="_blank" style="${linkStyle}">Walmart</a>
              <a href="https://www.instacart.com/store/search_v3/term?term=${q}" target="_blank" style="${linkStyle}">Instacart</a>
              <a href="https://www.barcodelookup.com/${upc}" target="_blank" style="${linkStyle}">Barcode Lookup</a>
            </div>
          </details>

          <div style="display:flex;align-items:center;gap:8px;border-top:1px solid #eee;padding-top:14px;">
            <span style="font-size:12px;color:#555;white-space:nowrap;">Labels:</span>
            <button id="pm-r-qty-dec" style="padding:4px 10px;font-size:15px;border:1px solid #ccc;border-radius:4px;background:#f5f5f5;cursor:pointer;">−</button>
            <input id="pm-r-qty" type="number" min="1" max="999" value="${this.printQty}"
              style="width:48px;text-align:center;padding:4px;font-size:14px;border:1px solid #ccc;border-radius:4px;">
            <button id="pm-r-qty-inc" style="padding:4px 10px;font-size:15px;border:1px solid #ccc;border-radius:4px;background:#f5f5f5;cursor:pointer;">+</button>
            <button id="pm-research-print"
              style="flex:1;padding:9px 0;background:#28a745;color:white;border:none;border-radius:5px;cursor:pointer;font-size:14px;font-weight:600;">
              🖨️ Print Label
            </button>
          </div>
        </div>
      `;

      document.body.appendChild(modal);

      // ── close ──
      modal.querySelector('#pm-research-close').addEventListener('click', () => modal.remove());
      modal.addEventListener('click', (e) => { if (e.target === modal) modal.remove(); });

      // ── qty stepper (mirrors main panel qty) ──
      const qtyInput = modal.querySelector('#pm-r-qty');
      modal.querySelector('#pm-r-qty-dec').addEventListener('click', () => {
        qtyInput.value = Math.max(1, parseInt(qtyInput.value||'1',10) - 1);
      });
      modal.querySelector('#pm-r-qty-inc').addEventListener('click', () => {
        qtyInput.value = Math.min(999, parseInt(qtyInput.value||'1',10) + 1);
      });

      // ── live margin calculator ──
      const priceInput = modal.querySelector('#pm-r-price');
      const costInput  = modal.querySelector('#pm-r-cost');
      const marginEl   = modal.querySelector('#pm-r-margin');
      const recalcMargin = () => {
        const p = parseFloat(priceInput.value);
        const c = parseFloat(costInput.value);
        if (!isNaN(p) && p > 0 && !isNaN(c) && c >= 0) {
          const margin = ((p - c) / p * 100).toFixed(1);
          const markup = ((p - c) / (c || 1) * 100).toFixed(1);
          const color  = margin >= 30 ? '#198754' : margin >= 15 ? '#856404' : '#dc3545';
          marginEl.innerHTML = `<span style="color:${color};font-weight:600;">Margin ${margin}%</span> &nbsp;·&nbsp; Markup ${markup}%`;
        } else {
          marginEl.textContent = '';
        }
      };
      priceInput.addEventListener('input', recalcMargin);
      costInput.addEventListener('input', recalcMargin);

      // ── AI pricing: factor input recalculates suggested price ──
      const factorInput  = modal.querySelector('#pm-r-factor');
      const aiPriceSpan  = modal.querySelector('#pm-r-ai-price');
      const marketRowEl  = modal.querySelector('#pm-r-market-row');
      let _marketHigh    = null;
      const recalcSuggested = () => {
        if (_marketHigh == null) return;
        const f = parseFloat(factorInput.value) || factor;
        const suggested = (_marketHigh * f).toFixed(2);
        aiPriceSpan.textContent = `$${suggested}`;
        return suggested;
      };
      factorInput.addEventListener('input', recalcSuggested);
      modal.querySelector('#pm-r-apply-ai').addEventListener('click', () => {
        const v = recalcSuggested();
        if (v) { priceInput.value = v; recalcMargin(); }
      });

      // ── print button ──
      modal.querySelector('#pm-research-print').addEventListener('click', async () => {
        const name  = modal.querySelector('#pm-r-name').value.trim();
        const price = priceInput.value.trim();
        const qty   = Math.max(1, parseInt(qtyInput.value||'1',10));
        if (!name)                     { alert('Please enter a product name.'); return; }
        if (!this.isValidPrice(price)) { alert('Please enter a valid price.'); return; }
        // Sync qty back to main panel
        this.printQty = qty;
        const panelQty = document.getElementById('pm-qty');
        if (panelQty) panelQty.value = qty;
        this._updateQtyLabel();
        // Save factor preference
        const f = parseFloat(factorInput.value);
        if (!isNaN(f) && f > 0) {
          this.printSettings.pricingFactor = f;
          this.savePrintSettingsToStorage();
        }
        modal.remove();
        await this.printCurrentProduct({ name, price, upc }, qty);
      });

      // ── async lookup — populates fields when data arrives ──
      this.lookupProductByUPC(upc).then(info => {
        const statusEl = document.getElementById('pm-r-lookup-status');
        if (!statusEl) return; // modal closed
        if (info) {
          statusEl.textContent = `✅ Found via ${info.source}`;
          const nameInput  = document.getElementById('pm-r-name');
          const brandInput = document.getElementById('pm-r-brand');
          const sizeInput  = document.getElementById('pm-r-size');
          if (nameInput  && info.name     && !nameInput.value)  nameInput.value  = info.name;
          if (brandInput && info.brand    && !brandInput.value) brandInput.value = info.brand;
          if (sizeInput  && info.quantity && !sizeInput.value)  sizeInput.value  = info.quantity;

          // AI pricing row
          if (info.priceMin != null || info.priceMax != null) {
            _marketHigh = info.priceMax ?? info.priceMin;
            const rangeEl = document.getElementById('pm-r-market-range');
            if (rangeEl) {
              rangeEl.textContent = (info.priceMin != null && info.priceMax != null)
                ? `$${info.priceMin.toFixed(2)} – $${info.priceMax.toFixed(2)}`
                : `$${(_marketHigh).toFixed(2)}`;
            }
            if (marketRowEl) marketRowEl.style.display = 'block';
            recalcSuggested();
          }
        } else {
          statusEl.textContent = '⚠️ Not found in product databases — fill in manually';
        }
      }).catch(() => {
        const statusEl = document.getElementById('pm-r-lookup-status');
        if (statusEl) statusEl.textContent = '⚠️ Lookup failed — fill in manually';
      });
    }

    toggleAutoPrint() {
      this.autoPrintEnabled = !this.autoPrintEnabled;
      const button = document.getElementById("pm-toggle-autoprint");
      if (button) {
        button.textContent = `Auto-Print: ${this.autoPrintEnabled ? "ON" : "OFF"}`;
        button.style.background = this.autoPrintEnabled ? "#28a745" : "#6c757d";
      }

      this.updateStatus(
        `Auto-print ${this.autoPrintEnabled ? "enabled" : "disabled"}`,
      );
      log(`🖨️ Auto-print ${this.autoPrintEnabled ? "enabled" : "disabled"}`);

      // Store preference
      try {
        localStorage.setItem(
          "pm-autoprint-enabled",
          this.autoPrintEnabled.toString(),
        );
        log(`💾 Auto-print preference saved: ${this.autoPrintEnabled}`);
      } catch (error) {
        log(`❌ Could not save auto-print preference:`, error);
      }
    }

    async checkPrintCapabilities() {
      try {
        const printers = await this.listBrowserPrintPrinters();
        if (printers.length > 0) {
          log("✅ Browser Print agent found printers:", printers.map(p => p.name));
          this.updateStatus(`✅ Printer ready (${printers[0].name || "unknown"})`, "success");
        } else {
          this.updateStatus("⚠️ Browser Print agent running but no printers found");
        }
      } catch (error) {
        log("⚠️ Browser Print agent not reachable:", error.message);
        this.updateStatus("⚠️ Zebra agent not reachable - is Browser Print running?");
      }
    }

    async testPrintServer() {
      this.updateStatus("Testing printer connection...");

      try {
        const testData = {
          name: "CONNECTION TEST",
          price: "$99.99",
          upc: "999999999999",
        };

        log("🧪 Testing printer with data:", testData);
        await this.printViaZebra(testData);
        this.updateStatus("✅ Printer test successful!", "success");
      } catch (error) {
        this.updateStatus("❌ Printer test failed: " + error.message, "error");
        log("Printer test failed:", error);
        await this.checkPrintCapabilities();
      }
    }

    showMissedItems() {
      const existing = document.getElementById("pm-missed-modal");
      if (existing) { existing.remove(); return; }

      const modal = document.createElement("div");
      modal.id = "pm-missed-modal";
      modal.style.cssText = "position:fixed;top:0;left:0;width:100%;height:100%;background:rgba(0,0,0,0.5);z-index:99999;display:flex;align-items:center;justify-content:center;";

      const renderRows = () => {
        if (!this.missedItems.length) return `<p style="color:#888;text-align:center;">No missed items.</p>`;
        return this.missedItems.map((m, i) => {
          const d = new Date(m.timestamp);
          const ts = d.toLocaleDateString() + " " + d.toLocaleTimeString([], {hour:"2-digit",minute:"2-digit"});
          return `<div style="display:flex;align-items:center;gap:8px;padding:6px 0;border-bottom:1px solid #eee;">
            <span style="flex:1;font-family:monospace;font-size:13px;">${m.upc}</span>
            <span style="font-size:11px;color:#999;white-space:nowrap;">${ts}</span>
            <button data-lookup="${m.upc}" style="font-size:11px;padding:3px 8px;cursor:pointer;">Search</button>
            <button data-remove="${i}" style="font-size:11px;padding:3px 8px;cursor:pointer;color:#c0392b;">✕</button>
          </div>`;
        }).join("");
      };

      modal.innerHTML = `
        <div style="background:white;border-radius:8px;padding:20px;width:420px;max-height:80vh;overflow-y:auto;position:relative;box-shadow:0 4px 20px rgba(0,0,0,0.3);">
          <button id="pm-missed-close" style="position:absolute;top:10px;right:12px;background:none;border:none;font-size:22px;cursor:pointer;color:#888;">&times;</button>
          <h3 style="margin:0 0 4px;">⚠️ Missed Items</h3>
          <p style="margin:0 0 12px;font-size:12px;color:#777;">Items scanned but not found in system. Click Lookup to open the new item flow.</p>
          <div id="pm-missed-list">${renderRows()}</div>
          <div style="margin-top:12px;text-align:right;">
            <button id="pm-missed-clear" style="font-size:11px;padding:4px 10px;cursor:pointer;color:#c0392b;">Clear All</button>
          </div>
        </div>`;

      document.body.appendChild(modal);

      modal.querySelector("#pm-missed-close").addEventListener("click", () => modal.remove());
      modal.addEventListener("click", e => { if (e.target === modal) modal.remove(); });

      modal.querySelector("#pm-missed-clear").addEventListener("click", () => {
        this.missedItems = [];
        this.saveMissedItems();
        this.updateMissedBadge();
        modal.querySelector("#pm-missed-list").innerHTML = renderRows();
      });

      modal.addEventListener("click", e => {
        if (e.target.dataset.lookup) {
          window.open(`https://www.google.com/search?q=${encodeURIComponent(e.target.dataset.lookup)}`, '_blank');
        }
        if (e.target.dataset.remove !== undefined) {
          const idx = parseInt(e.target.dataset.remove, 10);
          this.missedItems.splice(idx, 1);
          this.saveMissedItems();
          this.updateMissedBadge();
          modal.querySelector("#pm-missed-list").innerHTML = renderRows();
        }
      });
    }
